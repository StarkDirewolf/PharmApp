SELECT p.DrugDescription, d.ActualProductName FROM ProScriptConnect.ETP.EtpSummaryItemView p
JOIN PKBRuntime.Dmd.ActualProduct d ON p.DrugIdentifier = d.ActualProductId OR p.DrugIdentifier = d.VirtualProductId

ORDER BY p.PrescriptionId DESC

SELECT i.DispensedDate, ISNULL(c.Term, x.Description) FROM ProScriptConnect.PMR.PrescriptionItemDispensed i
LEFT JOIN PKBRuntime.Pharmacy.Pack p ON i.PackCodeId = p.PackCodeId
LEFT JOIN Coding.Code c ON c.CodeId = p.PreparationCodeId
LEFT JOIN ProScriptConnect.dbo.OwnDrug x ON i.PackCodeId = x.OwnDrugId
WHERE c.Term IS NULL
ORDER BY i.DispensedDate DESC
